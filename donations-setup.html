<script>
document.addEventListener("DOMContentLoaded", function () {

  const miniSection = document.getElementById("mini-bar-section");
  if (!miniSection) return;

  let miniLoaded = false;

  const loadMiniThermo = () => {
    if (miniLoaded) return;
    miniLoaded = true;

    /* ‚ùó Replace with YOUR OWN Script Web App URL */
    const BRT_MINI_URL =
      "YOUR_SCRIPT_URL_HERE?function=getDonations";

    const fill = document.getElementById("brt-mini-bar-fill");
    const valEl = document.getElementById("brt-mini-val");

    fetch(BRT_MINI_URL)
      .then(r => r.json())
      .then(data => {
        const total = Number(data.total || 0);
        const target = Number(data.target || 0);
        const pct = target > 0 ? Math.min(100, (total/target)*100) : 0;

        requestAnimationFrame(() => { fill.style.width = pct + "%"; });

        valEl.textContent =
          `¬£${total.toLocaleString("en-GB")} of ¬£${target.toLocaleString("en-GB")}`;
      });
  };

  new IntersectionObserver(entries => {
    if (entries[0].isIntersecting) {
      loadMiniThermo();
    }
  }, { rootMargin: "200px" }).observe(miniSection);

});
</script>

Replace: YOUR_SCRIPT_URL_HERE with the same Apps Script URL you used for GPX.

<script>
document.addEventListener("DOMContentLoaded", function () {

  const thermoSection = document.getElementById("main-thermo-section");
  if (!thermoSection) return;

  let thermoLoaded = false;

  const initThermometer = () => {
    if (thermoLoaded) return;
    thermoLoaded = true;

    /* ‚ùó Replace with YOUR OWN Script Web App URL */
    const BRT_THERMO_URL =
      "YOUR_SCRIPT_URL_HERE?function=getDonations";

    // Snow layer generator
    const snowLayer = document.getElementById("brt-thermo-snow");
    if (snowLayer) {
      for (let i = 0; i < 24; i++) {
        const s = document.createElement("span");
        s.className = "brt-snowflake";
        s.textContent = "‚Ä¢";
        s.style.left = Math.random() * 100 + "%";
        s.style.animationDuration = (4 + Math.random()*4) + "s";
        s.style.animationDelay = (Math.random()*4) + "s";
        snowLayer.appendChild(s);
      }
    }

    fetch(BRT_THERMO_URL)
      .then(r => r.json())
      .then(data => {
        const fill = document.getElementById("brt-thermo-fill");
        const amount = document.getElementById("brt-thermo-amount");
        const last = document.getElementById("brt-thermo-last-updated");

        const total = Number(data.total || 0);
        const target = Number(data.target || 0);
        const pct = Math.min(100, (total / target) * 100);

        requestAnimationFrame(() => {
          fill.style.height = pct + "%";
        });

        amount.textContent =
          `¬£${total.toLocaleString("en-GB")} raised of ¬£${target.toLocaleString("en-GB")} goal`;

        last.textContent =
          "Last updated: " + (data.lastUpdatePretty || "Not yet updated");
      })
      .catch(() => {
        const err = document.getElementById("brt-thermo-error");
        if (err) err.textContent = "Couldn't load donation total.";
      });
  };

  new IntersectionObserver(entries => {
    if (entries[0].isIntersecting) initThermometer();
  }, { rootMargin: "400px" }).observe(thermoSection);

});
</script>

Replace: YOUR_SCRIPT_URL_HERE with your own Apps Script deployment URL.
